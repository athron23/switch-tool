function renderPorts() { const top = document.getElementById("portsTop"); const bottom = document.getElementById("portsBottom"); top.innerHTML = ""; bottom.innerHTML = ""; const ports = switches[currentIndex].ports; const half = ports.length / 2; // 🔥 ปรับ column ตามจำนวน port const columnCount = half; top.style.gridTemplateColumns = `repeat(${columnCount}, 1fr)`; bottom.style.gridTemplateColumns = `repeat(${columnCount}, 1fr)`; ports.forEach(p => { const div = document.createElement("div"); div.className = "port " + getColor(p); // ✨ เพิ่ม label แบบ Cisco div.innerHTML = `
${p.id}
Gi1/0/${p.id}
`; div.onclick = () => showPort(p.id); if (p.id <= half) top.appendChild(div); else bottom.appendChild(div); }); }
